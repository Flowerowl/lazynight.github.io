偶然发现的有趣的两位数交换代码，来玩玩~~<!--more-->

1。

<pre class="lang:default decode:true">#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
int main(void){
    int a = 1;
    int b = 2;
    //a=     00000001
    //b=     00000010
    //(a^=b)=00000011
    //(b^=a)=00000001
    //(a^=b)=00000010
    a ^= b;
    b ^= a;
    a ^= b;
    printf("a=%d b=%d\n", a,b);
    system("PAUSE");
    return 0;
}</pre>
